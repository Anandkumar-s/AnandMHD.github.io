# spatial derivative function using finite difference method

def ddBrdz(B,dz,Nz):
    Br = np.zeros((Nz))
    for i in range(1,Nz-2):
        #Br[i]=(-B[i-2,0]+16*B[i-1,0]-30*B[i,0]+16*B[i+1,0]-B[i+2,0])/(12*dz*dz)
        Br[i]=(B[i-1,0]+(-2*B[i,0])+B[i+1,0])/(dz*dz)
    i=0
    Br[i]=((2*B[i,0])-(5*B[i+1,0])+(4*B[i+2,0])-B[i+3,0])/(dz*dz)
    i=Nz-1
    Br[i]=((2*B[i,0])-(5*B[i-1,0])+(4*B[i-2,0])-B[i-3,0])/(dz*dz)
    #print(B)
    return Br 
def dBrdz(B,dz,Nz):
    Br = np.zeros((Nz))

    for i in range(1,Nz-2):
        Br[i]=(B[i+1,0]-B[i-1,0])/(2*dz)
    i=0    
    Br[i]=(4*B[i+1,0]-B[i+2,0]-3*B[i,0])/(2*dz)
    i=Nz-1
    Br[i]=(-4*B[i-1,0]+B[i-2,0]+3*B[i,0])/(2*dz)
    return Br  

def ddBphidz(B,dz,Nz):
    Bphi = np.zeros((Nz))
    for i in range(1,Nz-2):
        #Bphi[i]=(-B[i-2,1]+16*B[i-1,1]-30*B[i,1]+16*B[i+1,1]-B[i+2,1])/(12*dz*dz)
        Bphi[i]=(B[i-1,1]+(-2*B[i,1])+B[i+1,1])/(dz*dz)
    i=0    
    Bphi[i]=((2*B[i,1])-(5*B[i+1,1])+(4*B[i+2,1])-B[i+3,1])/(dz*dz)
    i=Nz-1
    Bphi[i]=((2*B[i,1])-(5*B[i-1,1])+(4*B[i-2,1])-B[i-3,1])/(dz*dz)
    return Bphi 
 
def dBphidz(B,dz,Nz):
    Bphi = np.zeros((Nz))
    for i in range(1,Nz-2):
        Bphi[i]=(B[i+1,1]-B[i-1,1])/(2*dz)
    i=0    
    Bphi[i]=(4*B[i+1,1]-B[i+2,1]-3*B[i,1])/(2*dz)
    i=Nz-1
    Bphi[i]=(-4*B[i-1,1]+B[i-2,1]+3*B[i,1])/(2*dz)
    return Bphi
